sudo kill $(sudo lsof -t -i:8080)

kubectl create deployment rps --image cameronmcnz/roshambo


*** save as rps-service.yaml ***

apiVersion: v1
kind: Service
metadata:
  name: rps-service
spec:
  selector:
    app: rps
  ports:
  - port: 8080
    protocol: TCP
  type: NodePort
  externalTrafficPolicy: Cluster
  
kubectl apply -f rps-service.yaml

kubectl scale deployment --replicas=3 rps
