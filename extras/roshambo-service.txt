kubectl create deployment rps --image cameronmcnz/roshambo


apiVersion: v1
kind: Service
metadata:
  name: rps-service
spec:
  selector:
    app: rps
  ports:
  - port: 8080
    protocol: TCP
  type: NodePort
  externalTrafficPolicy: Cluster
  
kubectl apply -f rps-service.yaml

kubectl get services rps-service

minikube ip

http://<minikube ip>:<rps-service port>

kubectl scale deployment --replicas=3 rps
